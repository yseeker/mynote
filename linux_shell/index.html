    # {}の意味、https://qiita.com/yohm/items/3527d517768402efbcb6
seq 5 | awk '{print $1%2 ? "odd":"even"}'
# -c オプションは同じ行が何個連続しているかを数える
seq 5 | awk '{print $1%2 ? "odd":"even"}' | sort | uniq -c

2

    time ls *.png | sed 's/\.exe$//' | xargs -P$(nproc) -I@ convert @.png
# \はエスケープ文字.今は拡張子を検索したい。$は終端を表す。
# ファイルの中から.exeという名前のものを探す
